Part D

dt = .01;
t = -.5:dt:.5;
x = [ones(1,20) zeros(1,61) ones(1,20)];
x = x/(sum(x) * dt );
plot(t,x)
xlabel('time (s)');
ylabel('Amplitude');
title('Step input');
sigma = .4091;
minN = 0;
y = x;
ty = t;
g = 1/(sqrt(2*pi)*sigma)*exp(-(ty.^2)/(2*sigma^2));
 for i=1:1000
    [y,ty] = nconv(y,ty,x,t);
    y2 = sqrt(i)* y;
    t2 = ty / sqrt(i);
    g = 1/(sqrt(2*pi)*sigma)*exp(-(t2.^2)/(2*sigma^2));
    if max(y2 - g) < .01
        minN = i;
        break
    end
 end
figure();
plot(t2,g,t2,y2);
legend('normal function','convolution output');
xlabel('time(s)');
ylabel('Amplitude');

% y2 is also converging to a normal function
% Converges after 52 iterations
